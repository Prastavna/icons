# Icon Gallery

Browse all exported score symbols by category. Each card shows the outline and filled variant generated by the actual source exports.

  filled?: IconFunction

type IconModule = Record<string, IconFunction>

const query = ref('')
const copiedIconKey = ref('')

function formatLabel(value: string): string {
  return value
    .replace(/([A-Z])/g, ' $1')
    .replace(/^./, (char) => char.toUpperCase())
}

function buildPairs(iconModule: IconModule): IconPair[] {
  const pairs = new Map<string, IconPair>()

  for (const [exportName, icon] of Object.entries(iconModule)) {
    const key = exportName.replace(/Filled$/, '')
    const existing = pairs.get(key) ?? {
      key,
      label: formatLabel(key),
      searchText: '',
    }

    if (exportName.endsWith('Filled')) {
      existing.filled = icon
    } else {
      existing.outline = icon
    }

    pairs.set(key, existing)
  }

  return [...pairs.values()]
    .map((icon) => ({
      ...icon,
      searchText: `${icon.key} ${icon.label}`.toLowerCase(),
    }))
    .sort((a, b) => a.label.localeCompare(b.label))
}

const sections = categories.map((category) => ({
  ...category,
  icons: buildPairs(category.module),
}))

const filteredSections = computed(() => {
  const term = query.value.trim().toLowerCase()

  if (!term) {
    return sections
  }

  return sections
    .map((section) => ({
      ...section,
      icons: section.icons.filter(
        (icon) => icon.searchText.includes(term) || section.title.toLowerCase().includes(term),
      ),
    }))
    .filter((section) => section.icons.length > 0)
})

const totalIcons = sections.reduce((count, section) => count + section.icons.length, 0)
const filteredIconCount = computed(() =>
  filteredSections.value.reduce((count, section) => count + section.icons.length, 0),
)

async function copyIconName(name: string) {
  if (typeof navigator === 'undefined' || !navigator.clipboard) {
    return
  }

  await navigator.clipboard.writeText(name)
  copiedIconKey.value = name

  window.setTimeout(() => {
    if (copiedIconKey.value === name) {
      copiedIconKey.value = ''
    }
  }, 1500)
}
