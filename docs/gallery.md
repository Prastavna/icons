# Icon Gallery

Browse all exported score symbols by category. Each card shows the outline and filled variant generated by the actual source exports.

<script setup lang="ts">
import * as notes from '../src/icons/notes/index.ts'
import * as rests from '../src/icons/rests/index.ts'
import * as clefs from '../src/icons/clefs/index.ts'
import * as accidentals from '../src/icons/accidentals/index.ts'
import * as timeSignatures from '../src/icons/time-signatures/index.ts'
import * as dynamics from '../src/icons/dynamics/index.ts'
import * as articulations from '../src/icons/articulations/index.ts'
import * as ornaments from '../src/icons/ornaments/index.ts'
import * as repeats from '../src/icons/repeats/index.ts'
import * as barlines from '../src/icons/barlines/index.ts'
import * as beams from '../src/icons/beams/index.ts'
import * as ties from '../src/icons/ties/index.ts'
import * as slurs from '../src/icons/slurs/index.ts'
import * as pedals from '../src/icons/pedals/index.ts'
import * as bows from '../src/icons/bows/index.ts'
import * as misc from '../src/icons/misc/index.ts'

type IconFunction = (options?: Record<string, string | number>) => string

type IconPair = {
  key: string
  label: string
  outline?: IconFunction
  filled?: IconFunction
}

type IconModule = Record<string, IconFunction>

const categories = [
  { title: 'Notes', path: '@prastavna/icons/notes', module: notes },
  { title: 'Rests', path: '@prastavna/icons/rests', module: rests },
  { title: 'Clefs', path: '@prastavna/icons/clefs', module: clefs },
  { title: 'Accidentals', path: '@prastavna/icons/accidentals', module: accidentals },
  { title: 'Time Signatures', path: '@prastavna/icons/time-signatures', module: timeSignatures },
  { title: 'Dynamics', path: '@prastavna/icons/dynamics', module: dynamics },
  { title: 'Articulations', path: '@prastavna/icons/articulations', module: articulations },
  { title: 'Ornaments', path: '@prastavna/icons/ornaments', module: ornaments },
  { title: 'Repeats', path: '@prastavna/icons/repeats', module: repeats },
  { title: 'Barlines', path: '@prastavna/icons/barlines', module: barlines },
  { title: 'Beams', path: '@prastavna/icons/beams', module: beams },
  { title: 'Ties', path: '@prastavna/icons/ties', module: ties },
  { title: 'Slurs', path: '@prastavna/icons/slurs', module: slurs },
  { title: 'Pedals', path: '@prastavna/icons/pedals', module: pedals },
  { title: 'Bows', path: '@prastavna/icons/bows', module: bows },
  { title: 'Misc', path: '@prastavna/icons/misc', module: misc },
]

function formatLabel(value: string): string {
  return value
    .replace(/([A-Z])/g, ' $1')
    .replace(/^./, (char) => char.toUpperCase())
}

function buildPairs(iconModule: IconModule): IconPair[] {
  const pairs = new Map<string, IconPair>()

  for (const [exportName, icon] of Object.entries(iconModule)) {
    const key = exportName.replace(/Filled$/, '')
    const existing = pairs.get(key) ?? {
      key,
      label: formatLabel(key),
    }

    if (exportName.endsWith('Filled')) {
      existing.filled = icon
    } else {
      existing.outline = icon
    }

    pairs.set(key, existing)
  }

  return [...pairs.values()].sort((a, b) => a.label.localeCompare(b.label))
}

const sections = categories.map((category) => ({
  ...category,
  icons: buildPairs(category.module),
}))
</script>

<p class="muted">
  Tip: use root exports like <code>noteQuarter</code> when you want one import surface, or category subpaths like
  <code>@prastavna/icons/notes</code> when you want shorter names.
</p>

<div class="icon-gallery">
  <section v-for="section in sections" :key="section.title">
    <h2>{{ section.title }}</h2>
    <p class="muted">
      {{ section.icons.length }} icons per variant. Import from <code>{{ section.path }}</code>.
    </p>
    <div class="icon-category-grid">
      <article v-for="icon in section.icons" :key="icon.key" class="icon-card">
        <header>
          <h3>{{ icon.label }}</h3>
          <code>{{ icon.key }}</code>
        </header>
        <div class="icon-preview-row">
          <div class="icon-preview">
            <div v-if="icon.outline" v-html="icon.outline({ size: 40 })" />
            <strong>Outline</strong>
          </div>
          <div class="icon-preview">
            <div v-if="icon.filled" v-html="icon.filled({ size: 40 })" />
            <strong>Filled</strong>
          </div>
        </div>
      </article>
    </div>
  </section>
</div>
